神经网络的反向传播算法

一、前置知识

1.Logistic回归算法

目标：将给定的数据进行分类

输入：$ ^n \(，输出：\) y $

模型：$ P() = sigmoid(w^Tx+b), w^n, x^n, b $

模型的训练:通过极大化对数似然函数：

\[ \begin{aligned} -L(\pmb{w}) = \frac 1 m \sum_{i = 1}^m y^ilog(P(\pmb{x}^i)) + (1-y^i)log(1-P(\pmb{x}^i)) \end{aligned} \]

即最小化：

\[ \begin{aligned} L(\pmb{w}) = -\frac 1 m \sum_{i = 1}^m y^ilog(P(\pmb{x}^i)) + (1-y^i)log(1-P(\pmb{x}^i)) \end{aligned} \]

\[ \begin{aligned} \frac{\partial}{\partial \pmb{w}}L(\pmb{w}) = \frac 1 m \sum_{i = 1}^m(P(\pmb{x^i}) - y^i)\pmb{x^i} \end{aligned} \]

\[ \begin{aligned} \frac{\partial}{\partial b}L(\pmb{w}) = \frac 1 m \sum_{i = 1}^mP(\pmb{x^i})-y^i \end{aligned} \]

更新参数：

\[ \begin{aligned} \pmb{w} = \pmb{w} - \alpha \frac{\partial}{\partial \pmb{w}}L(\pmb{w}) \end{aligned} \]

\[ \begin{aligned} b = b - \alpha \frac{\partial}{\partial b}L(\pmb{w}) \end{aligned} \]

二、反向传播算法

1.标量形式推导

\[ \begin{aligned} L = -\frac 1 m \sum_{k = 1}^m y_klog(a^2_k) + (1-y_k)log(1-a^2_k)) \end{aligned} \]

\[ \begin{aligned} \\ \frac{\partial L}{\partial W^2_{ij}} &= \frac{\partial L}{\partial z^2_i} \frac{\partial z^2_i}{\partial W^2_{ij}} \\ &= \frac{\partial L}{\partial a^2_i}\frac{\partial a^2_i}{\partial z^2_i}\frac{\partial z^2_i}{\partial W^2_{ij}} \\ &= -(\frac{y_i}{a^2_i}+\frac{1-y_i}{1-a^2_i})a^2_i(1 - a^2_i)a^1_j \\ &= (a^2_i-y_i)a^1_j \\ \\ \frac{\partial L}{\partial W^1_{ij}} &= ... \end{aligned} \]

2.矩阵形式推导

链式求导法

\[ \begin{aligned} L &= \sum_{i,j}-\pmb{Y_{ij}}\log \pmb{A^2_{ij}} - \sum_{i,j}(1-\pmb{Y_{ij}})\log (1-\pmb{A^2_{ij}})\\ \\ \frac{\partial L}{\partial \pmb{Z^2_{ij}}} &= -\pmb{Y_{ij}}(1-\pmb{A^2_{ij}}) - (1-\pmb{Y_{ij}})\frac{1}{1-\pmb{A^2_{ij}}}\pmb{A_{ij}}(1-\pmb{A^2_{ij}})\\ \frac{\partial L}{\partial \pmb{Z^2}} &= -\pmb{Y}\odot (1-\pmb{A^2}) - (1-\pmb{Y})\odot \pmb{A^2}\\ &= \pmb{A^2} - \pmb{Y}\\ \frac{\partial L}{\partial \pmb{W_2}} &=\frac{\partial L}{\partial \pmb{Z^2}}\frac{\partial \pmb{Z^2}}{\partial \pmb{W^2}}\\ &=(\pmb{A^2} - \pmb{Y}) \end{aligned} \]

微分法

相关运算性质：

\[ \begin{aligned} &tr(A^T(B\odot C)) = tr((A\odot B)^TC) = \sum_{i,j}A_{ij}B_{ij}C_{ij}\\ &d\sigma (Z) = \sigma (Z)\odot (1-\sigma (Z))dZ\\ &d(AB)=(dA)B + AdB\\ &d(c_1A+c_2B) = c_1dA+c_2dB\\ &dA^T = (dA)^T\\ &tr(A^TB) = \sum_{i,j}A_{ij}B_{ij} \end{aligned} \]

\[ \begin{aligned} \pmb{L} &= -\pmb{Y}\odot \log \pmb{A^2} - (1 - \pmb{Y})\odot \log (1 - \pmb{A^2}) \\ L &= tr(-\pmb{Y^T}\log \pmb{A^2}) - tr((1-\pmb{Y})^T\log (\pmb{1-A^2})) \\ \\ dL &= tr(d(-\pmb{Y^T}\log \pmb{A^2})) - tr(d((1 - \pmb{Y})^T\log (1 - \pmb{A^2}))) \\ &= tr(-d(\pmb{Y^T})\log \pmb{A^2} - \pmb{Y^T}d\log \pmb{A^2}) - tr(d((1 - \pmb{Y})^T)\log (1-\pmb{A^2})+(1-\pmb{Y})^Td\log (1-\pmb{A^2}))\\ &= tr(-\pmb{Y^T}(\frac{1}{\pmb{A^2}}\odot d\pmb{A^2})) + tr((1-\pmb{Y})^T\frac{1}{1-\pmb{A^2}}\odot d\pmb{A^2})\\ &=tr((1-\pmb{Y})^T\frac{1}{1-\pmb{A^2}}\odot d\pmb{A^2}) -tr(\pmb{Y^T}(\frac{1}{\pmb{A^2}}\odot d\pmb{A^2}))\\ &=tr(((1-\pmb{Y})\odot \frac{1}{1-\pmb{A^2}})^Td\pmb{A^2}) - tr((\pmb{Y}\odot \frac{1}{\pmb{A^2}})^Td\pmb{A^2})\\ &=tr(((1-\pmb{Y})\odot \frac{1}{1-\pmb{A^2}}\odot \pmb{A^2}\odot(1 - \pmb{A^2}))^T d\pmb{Z^2}) - tr((\pmb{Y}\odot \frac{1}{\pmb{A^2}}\odot \pmb{A^2}\odot(1 - \pmb{A^2}))^Td\pmb{Z^2})\\ &=tr(((1-\pmb{Y})\odot \pmb{A^2})^Td\pmb{Z^2}) - tr((\pmb{Y}\odot (1-\pmb{A^2}))^Td\pmb{Z^2})\\ &=tr((\pmb{A^2} - \pmb{Y})^Td\pmb{Z^2})\\ \\ &\frac{\partial L}{\partial \pmb{Z^2}} = \pmb{A^2} - \pmb{Y}\\ \\ \end{aligned} \]

\[ \begin{aligned} dL &= tr((\pmb{A^2} - \pmb{Y})^T(d\pmb{A^1})\pmb{(W^2)}^T) + tr((\pmb{A^2} - \pmb{Y})^T\pmb{A^1}d\pmb{(W^2)}^T)\\ &=tr(\pmb{(W^2)}^T(\pmb{A^2} - \pmb{Y})^Td\pmb{A^1}) + tr(((\pmb{A^2} - \pmb{Y})^T\pmb{A^1})^Td\pmb{W^2})\\ &=tr((\pmb{W^2})^T(\pmb{A^2}-\pmb{Y})^T\pmb{A^1}\odot (1-\pmb{A^1})\odot d\pmb{Z^1}) + tr(((\pmb{A^2} - \pmb{Y})^T\pmb{A^1})^Td\pmb{W^2})\\ &=tr(((\pmb{A^2} - \pmb{Y})\pmb{W^2})^T\pmb{A^1}\odot (1-\pmb{A^1})\odot d\pmb{Z^1}) + tr(((\pmb{A^2} - \pmb{Y})^T\pmb{A^1})^Td\pmb{W^2})\\ &=tr(((\pmb{A^2}-\pmb{Y})\pmb{W^2}\odot \pmb{A^1}\odot (1-\pmb{A^1}))^Td\pmb{Z^1}) + tr(((\pmb{A^2} - \pmb{Y})^T\pmb{A^1})^Td\pmb{W^2})\\ &=tr(((\pmb{A^2}-\pmb{Y})\pmb{W^2}\odot \pmb{A^1}\odot (1-\pmb{A^1}))^T((d\pmb{X})\pmb{W^1}^T+\pmb{X}d(\pmb{W^1})^T) + tr(((\pmb{A^2} - \pmb{Y})^T\pmb{A^1})^Td\pmb{W^2})\\ &=tr(((\pmb{A^2}-\pmb{Y})\pmb{W^2}\odot \pmb{A^1}\odot (1-\pmb{A^1}))^T\pmb{X}d(\pmb{W^1})^T)+tr(((\pmb{A^2} - \pmb{Y})^T\pmb{A^1})^Td\pmb{W^2})\\ &=tr((((\pmb{A^2}-\pmb{Y})\pmb{W^2}\odot \pmb{A^1}\odot (1-\pmb{A^1}))^T\pmb{X})^Td\pmb{W^1}) +tr(((\pmb{A^2} - \pmb{Y})^T\pmb{A^1})^Td\pmb{W^2})\\ \\ &\frac{\partial L}{\partial \pmb{W^2}} = (\pmb{A^2} - \pmb{Y})^T\pmb{A^1}\\ &\frac{\partial L}{\partial \pmb{W^1}} = ((\pmb{A^2}-\pmb{Y})\pmb{W^2}\odot \pmb{A^1}\odot (1-\pmb{A^1}))^T\pmb{X} \end{aligned} \]
